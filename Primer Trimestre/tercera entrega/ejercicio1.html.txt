        function addAnimal() {
            const ul = document.getElementById('listaelemen');
            const newAnimal = document.createElement('li');
            newAnimal.innerText = 'Nuevo Animal'; // Cambia "Nuevo Animal" por el nombre que desees
            ul.appendChild(newAnimal);
        }

        function insertAnimal() {
            const ul = document.getElementById('listaelemen');
            const newAnimal = document.createElement('li');
            newAnimal.innerText = 'Animal Insertado'; // Cambia por el nombre que desees
            ul.insertBefore(newAnimal, ul.children[0]); // Insertar antes del primer elemento
        }

        function replaceAnimal() {
            const ul = document.getElementById('listaelemen');
            if (ul.children.length > 1) {
                ul.children[1].innerText = 'Animal Reemplazado'; // Cambia por el nombre que desees
            }
        }

        function deleteAnimal() {
            const ul = document.getElementById('listaelemen');
            if (ul.children.length > 1) {
                ul.removeChild(ul.children[1]); // Eliminar el segundo animal
            }
        }

        function createSublistDOM() {
            const lion = document.querySelector('li:first-child');
            const sublist = document.createElement('ul');
            const subAnimal = document.createElement('li');
            subAnimal.innerText = 'Subanimal'; // Cambia por el nombre que desees
            sublist.appendChild(subAnimal);
            lion.appendChild(sublist);
        }

        function createSublistInnerHTML() {
            const lion = document.querySelector('li:first-child');
            lion.innerHTML += '<ul><li>Subanimal InnerHTML</li></ul>'; // Cambia por el nombre que desees
        }
